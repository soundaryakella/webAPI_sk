pipeline {

        stage('Checkout') {

                always {

                        alwaysLinkToLastBuild: true,

                    ])
                }
            }
        }

            steps {

            }

                    subject: "Test Results: ${env.JOB_NAME} - ${env.BUILD_NUMBER}",
                    body: "Build: ${env.BUILD_URL}\nResults: ${env.BUILD_URL}testReport/",
                    to: "team@company.com"

        }
